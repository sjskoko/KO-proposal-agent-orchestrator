테스트

단위·통합·에이전트·스킬 테스트를 실행하는 방법

테스트 계층

계층위치Ollama 필요목적
단위tests/unit/불필요인터페이스, 로직, 상태머신 검증
통합tests/integration/불필요런타임 실제 동작 검증
에이전트 평가tests/agent_eval/불필요 (Stub 모델)전체 흐름 시나리오
스킬 평가tests/skill_eval/불필요스킬 핸들러 단독 검증

실행 방법

# 전체 테스트
bash scripts/run_tests.sh all

# 계층별 실행
bash scripts/run_tests.sh unit
bash scripts/run_tests.sh integration
bash scripts/run_tests.sh agent
bash scripts/run_tests.sh skill

# pytest 직접 실행
pytest tests/ -v --tb=short
pytest tests/unit/ -v
pytest tests/unit/test_permissions.py -v -k "test_require_denied"

# 커버리지
pytest tests/ --cov=core --cov=runtimes --cov-report=html
open htmlcov/index.html

런타임 통합 테스트

# tests/integration/test_runtime_memory.py
import pytest
from core.runtime.base import RuntimeCall
from runtimes.memory.runtime import MemoryRuntime

@pytest.fixture
def runtime():
    rt = MemoryRuntime()
    rt.configure({"backend": "in_memory"})
    return rt

class TestMemoryRuntime:
    def test_write_and_read(self, runtime):
        # 쓰기
        write = runtime.execute(RuntimeCall("memory", "memory_write", {
            "content": "중요한 정보",
            "scope": "session",
            "agent_id": "test_agent"
        }))
        assert write.success

        # 읽기
        read = runtime.execute(RuntimeCall("memory", "memory_read", {
            "query": "중요한",
            "scope": "session",
            "agent_id": "test_agent"
        }))
        assert read.success
        assert len(read.data["entries"]) > 0

에이전트 평가 테스트

# tests/agent_eval/test_custom_agent.py
from core.agent.base import Agent
from core.agent.definition import AgentDefinition, ModelPolicy
from core.events.bus import EventBus
from core.events.types import AgentDoneEvent
from core.memory.base import MemoryScope
from core.permissions.base import PermissionSet

class _StubModel:
    name = "stub"
    def complete(self, messages, options=None):
        from core.model.base import ModelResponse
        # 항상 단일 단계 플랜 반환
        return ModelResponse(
            content='[{"description":"task","task_type":"reasoning"}]',
            model="stub", provider="stub"
        )
    def stream(self, m, o=None): yield "x"
    def health_check(self): return True

def test_my_agent_completes_goal():
    bus = EventBus()
    done_events = []
    bus.subscribe(AgentDoneEvent, done_events.append)

    defn = AgentDefinition(
        id="my_agent", role="tester", goal="test",
        model_policy=ModelPolicy(preferred="stub"),
        memory_scope=MemoryScope.SESSION,
        permissions=PermissionSet.unrestricted(),
    )
    from core.planner.sequential import SequentialPlanner
    agent = Agent(defn, bus, planner=SequentialPlanner(_StubModel()))
    agent.run("목표 달성")

    assert done_events[0].success
